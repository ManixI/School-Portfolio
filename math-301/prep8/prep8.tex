\begin{enumerate}
  \item The elements belong to set B, the elements also contain set A
  \item Proposition: $A \subset B$ where $A = \{a\in\mathbb{Z}:4|a\}$ and $B = \{b\in\mathbb{Z}:4|b\}$ \\
        Suppose $A = \{a\in\mathbb{Z}:4|a\}$\\
        This means $a\in\mathbb{Z}$ and $4|a$\\
        Then there exists $c$ where $a=4c$ and $c\in\mathbb{Z}$ by definition of divisibility\\
        Therefor $a=2(2c)$, showing $2|a$\\
        So because $a$ is an integer that 2 divides $a\in\{x\in\mathbb{Z}:2|z\}$\\
        Thus $A\subset B$ 
\end{enumerate}
